\section{hyperref}

\mintinline{latex}{\usepackage{hyperref}}

\subsection{Options}

\begin{minted}{latex}
\hypersetup{<options>}
\end{minted}

\begin{itemize}
  \item colorlinks (false): colors he text of links and anchors
  \item linkcolor (red): normal internal links
  \item anchorcolor (black): anchor text
  \item citecolor (green): bibliographical citations
  \item filecolor (cyan): local files
  \item menucolor (red): acrobat menu items
  \item runcolor (filecolor): run links (launch annotations)
  \item urlcolor (magenta): URLs
  \item allcolors
\end{itemize}

\subsection{Example}

\begin{minted}{latex}
\hypersetup{
  colorlinks
}
\end{minted}

\subsection{Additional user macros}

\begin{minted}{latex}
\href{URL}{text}
\url{URL}
\hyperref{URL}{category}{name}{text} % URL#category.name
\end{minted}
