\documentclass[../../latex]{subfiles}

\begin{document}

\section{listings}

\mintinline{latex}{\usepackage{listings}}

\subsection{Inline code}

\mintinline{latex}{\lstinline!<text>!} (``\mintinline{latex}{|}'' can be replaced by any character)

\begin{minipage}{0.47 \textwidth}
  \begin{minted}{latex}
\lstinline|Hello, world!|
  \end{minted}
\end{minipage}
\hfill
\begin{minipage}{0.47 \textwidth}
  \lstinline|Hello, world!|
\end{minipage}

\subsection{Code block}

\mintinline{latex}{\begin{lstlisting} ... \end{lstlisting}}

\begin{minipage}{0.47 \textwidth}
  \begin{minted}{latex}
\begin{lstlisting}
def hello():
    print("Hello, world!")
\end{lstlisting}
  \end{minted}
\end{minipage}
\hfill
\begin{minipage}{0.47 \textwidth}
  \begin{lstlisting}
def hello():
    print("Hello, world!")
  \end{lstlisting}
\end{minipage}

\subsection{Input file}

\mintinline{latex}{\lstinputlisting{<file-path>}}

\begin{minipage}{0.47 \textwidth}
  \begin{minted}{latex}
\lstinputlisting{hello.py}
  \end{minted}
\end{minipage}
\hfill
\begin{minipage}{0.47 \textwidth}
  \lstinputlisting{hello.py}
\end{minipage}

\end{document}
