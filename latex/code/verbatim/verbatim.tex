\section{verbatim}

\subsection{Inline code}

\mintinline{latex}{\verb|<text>|} (``\mintinline{latex}{|}'' can be replaced by any character except ``\mintinline{latex}{*}'')

\begin{minipage}{0.47 \textwidth}
  \begin{minted}{latex}
\verb|Hello, world!|
  \end{minted}
\end{minipage}
\hfill
\begin{minipage}{0.47 \textwidth}
  \verb|Hello, world!|
\end{minipage}

\subsection{Code block}

\mintinline{latex}{\begin{verbatim} ... \end{verbatim}}

\begin{minipage}{0.47 \textwidth}
  \begin{minted}{latex}
\begin{verbatim}
def hello():
    print("Hello, world!")
\end{verbatim}
  \end{minted}
\end{minipage}
\hfill
\begin{minipage}{0.47 \textwidth}
  \begin{verbatim}
def hello():
    print("Hello, world!")
  \end{verbatim}
\end{minipage}

\subsection{Block comment}

\mintinline{latex}{\begin{comment} ... \end{comment}}

\begin{minipage}{0.47 \textwidth}
  \begin{minted}{latex}
Text 1

\begin{comment}
This part will be ignored.
\end{comment}

Text 2
  \end{minted}
\end{minipage}
\hfill
\begin{minipage}{0.47 \textwidth}
  Text 1

  \begin{comment}
  This part will be ignored.
  \end{comment}

  Text 2
\end{minipage}
