\documentclass[../../latex]{subfiles}

\begin{document}

\section{parskip}

\mintinline{latex}{\usepackage[<options>]{parskip}}

Customize indentation and margins of paragraphs.

\subsection{Options}

\begin{itemize}
  \item skip: set \textbackslash parskip
  \item tocskip: set \textbackslash parskip within \textbackslash tableofcontents
  \item indent: set \textbackslash parindent
  \item parfill: set \textbackslash parfillskip
\end{itemize}

\subsection{Example}

\begin{minted}{latex}
\usepackage[
  skip=\baselineskip,
  indent=0pt,
]{parskip}
\end{minted}
