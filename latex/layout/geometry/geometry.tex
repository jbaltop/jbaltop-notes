\documentclass[../../latex]{subfiles}

\begin{document}

\section{geometry}

\mintinline{latex}{\usepackage{geometry}}

\begin{minted}{latex}
\usepackage[<options>]{geometry}
\end{minted}

or

\begin{minted}{latex}
\usepackage{geometry}

\geometry{<options>}
\end{minted}

\subsection{Paper size}

\begin{minted}{latex}
\geometry{
  paper=<paper-name>, % paper size
  screen=<(W,H)>, % paper size in width & height
  paperwidth=<length>,
  paperheight=<length>,
  papersize={<width>,<height>},
  landscape,
  portrait,
}
\end{minted}

\subsection{Margin size}

\begin{minted}{latex}
\geometry{
  left=<length>,
  inner=<length>,
  right=<length>,
  outer=<length>,
  top=<length>,
  bottom=<length>,
  hmargin=<length>, % left & right
  vmargin=<length>, % top & bottom
  margin=<length>, % hmargin & vmargin
}
\end{minted}

\subsection{Example}

\begin{minted}{latex}
\geometry{
  paper=a4paper,
  margin=15mm,
}
\end{minted}
