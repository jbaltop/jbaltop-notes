\section{Text over and under symbols}

\subsection{Place text using overset \& underset}

\begin{minted}{latex}
\usepackage{amsmath} % align

\begin{align}
  a \overset{why?}{=} b \\
  a \underset{why?}{=} b
\end{align}
\end{minted}

\begin{align}
  a \overset{why?}{=} b \\
  a \underset{why?}{=} b
\end{align}

\subsection{Remove extra spaces with mathclap}

\begin{minted}{latex}
\usepackage{amsmath} % align
\usepackage{mathtools} % mathclap

\begin{align}
  a \overset{
    \mathclap{why?}
  }{=} b \\
  a \underset{
    \mathclap{why?}
  }{=} b
\end{align}
\end{minted}

\begin{align}
  a \overset{
    \mathclap{why?}
  }{=} b \\
  a \underset{
    \mathclap{why?}
  }{=} b
\end{align}
