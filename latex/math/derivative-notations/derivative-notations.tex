\section{Derivative notations}

\mintinline{latex}{\usepackage{derivative}}

\subsection{Ordinary derivative}

\begin{minipage}{0.47 \textwidth}
  \begin{minted}{latex}
\begin{align}
  & \frac{df}{dx} \\
  & \odv{f}{x} \\
  & \odv*{f}{x}
\end{align}
  \end{minted}
\end{minipage}
\hfill
\begin{minipage}{0.47 \textwidth}
  \begin{align}
     & \frac{df}{dx} \\
     & \odv{f}{x}    \\
     & \odv*{f}{x}
  \end{align}
\end{minipage}

\subsection{Partial derivative}

\begin{minipage}{0.47 \textwidth}
  \begin{minted}{latex}
\begin{align}
  & \frac{\partial f}{\partial x} \\
  & \pdv{f}{x} \\
  & \pdv*{f}{x} \\
  & \pdv{f}!{x} \\
  & \pdv{f}{x,y}
  & \pdv[order={2,3}]{f}{x,y,z} \\
\end{align}
  \end{minted}
\end{minipage}
\hfill
\begin{minipage}{0.47 \textwidth}
  \begin{align}
     & \frac{\partial f}{\partial x} \\
     & \pdv{f}{x}                    \\
     & \pdv*{f}{x}                   \\
     & \pdv{f}!{x}                   \\
     & \pdv{f}{x,y}                  \\
     & \pdv[order={2,3}]{f}{x,y,z}
  \end{align}
\end{minipage}
